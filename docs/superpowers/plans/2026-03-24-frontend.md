```tsx
'use client'

import { createContext, useContext, useEffect, useState, type ReactNode } from 'react'
import { onAuthStateChanged, type User } from '@/lib/firebase'

interface AuthContextValue {
  user: User | null
  loading: boolean
}

const AuthContext = createContext<AuthContextValue>({ user: null, loading: true })

export function FirebaseAuthProvider({ children }: { children: ReactNode }): JSX.Element {
  const [user, setUser] = useState<User | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    const unsub = onAuthStateChanged(auth => {
      setUser(auth)
      setLoading(false)
    })
    return unsub
  }, [])

  return <AuthContext.Provider value={{ user, loading }}>{children}</AuthContext.Provider>
}

export function useAuth(): AuthContextValue {
  return useContext(AuthContext)
}
```

```tsx
'use client'
import React, { useState, useEffect } from 'react'
import { motion } from 'framer-motion'
import { cn } from '@/lib/utils'

type Direction = 'TOP' | 'LEFT' | 'BOTTOM' | 'RIGHT'

const movingMap: Record<Direction, string> = {
  TOP: 'radial-gradient(20.7% 50% at 50% 0%, hsl(0, 0%, 100%) 0%, rgba(255, 255, 255, 0) 100%)',
  LEFT: 'radial-gradient(16.6% 43.1% at 0% 50%, hsl(0, 0%, 100%) 0%, rgba(255, 255, 255, 0) 100%)',
  BOTTOM: 'radial-gradient(20.7% 50% at 50% 100%, hsl(0, 0%, 100%) 0%, rgba(255, 255, 255, 0) 100%)',
  RIGHT: 'radial-gradient(16.2% 41.2% at 100% 50%, hsl(0, 0%, 100%) 0%, rgba(255, 255, 255, 0) 100%)',
}

const highlight = 'radial-gradient(75% 181.16% at 50% 50%, #3275F8 0%, rgba(255, 255, 255, 0) 100%)'

export function HoverBorderGradient({
  children,
  containerClassName,
  className,
  as: Element = 'button',
  duration = 1,
  clockwise = true,
  ...props
}: React.PropsWithChildren<{
  as?: React.ElementType
  containerClassName?: string
  className?: string
  duration?: number
  clockwise?: boolean
} & React.HTMLAttributes<HTMLElement>>): JSX.Element {
  const [hovered, setHovered] = useState(false)
  const [direction, setDirection] = useState<Direction>('BOTTOM')

  const rotateDirection = (cur: Direction): Direction => {
    const dirs: Direction[] = ['TOP', 'LEFT', 'BOTTOM', 'RIGHT']
    const idx = dirs.indexOf(cur)
    return clockwise ? dirs[(idx - 1 + 4) % 4]! : dirs[(idx + 1) % 4]!
  }

  useEffect(() => {
    if (hovered) return
    const id = setInterval(() => setDirection(d => rotateDirection(d)), duration * 1000)
    return () => clearInterval(id)
  }, [hovered, duration, clockwise])

  return (
    <Element
      onMouseEnter={() => setHovered(true)}
      onMouseLeave={() => setHovered(false)}
      className={cn('relative flex h-min w-fit items-center justify-center overflow-visible rounded-full border bg-black/40 p-px backdrop-blur-sm transition duration-500', containerClassName)}
      {...props}
    >
      <div className={cn('z-10 rounded-[inherit] bg-black px-4 py-2 text-white', className)}>
        {children}
      </div>
      <motion.div
        className="absolute inset-0 z-0 overflow-hidden rounded-[inherit]"
        style={{ filter: 'blur(2px)', width: '100%', height: '100%' }}
        initial={{ background: movingMap[direction] }}
        animate={{ background: hovered ? [movingMap[direction], highlight] : movingMap[direction] }}
        transition={{ ease: 'linear', duration }}
      />
      <div className="absolute inset-0.5 z-[1] rounded-[100px] bg-black" />
    </Element>
  )
}
```

```tsx
'use client'
import { type FC, type ReactNode, useRef } from 'react'
import { motion, type MotionValue, useScroll, useTransform } from 'framer-motion'
import { cn } from '@/lib/utils'

interface TextRevealByWordProps { text: string; className?: string }

export const TextRevealByWord: FC<TextRevealByWordProps> = ({ text, className }) => {
  const targetRef = useRef<HTMLDivElement | null>(null)
  const { scrollYProgress } = useScroll({ target: targetRef })
  const words = text.split(' ')
  return (
    <div ref={targetRef} className={cn('relative z-0 h-[200vh]', className)}>
      <div className="sticky top-0 mx-auto flex h-[50%] max-w-4xl items-center bg-transparent px-[1rem] py-[5rem]">
        <p className="flex flex-wrap p-5 text-2xl font-bold text-white/20 md:p-8 md:text-3xl lg:p-10 lg:text-4xl xl:text-5xl">
          {words.map((word, i) => {
            const start = i / words.length
            const end = start + 1 / words.length
            return <Word key={i} progress={scrollYProgress} range={[start, end]}>{word}</Word>
          })}
        </p>
      </div>
    </div>
  )
}

const Word: FC<{ children: ReactNode; progress: MotionValue<number>; range: [number, number] }> = ({ children, progress, range }) => {
  const opacity = useTransform(progress, range, [0, 1])
  return (
    <span className="xl:lg-3 relative mx-1 lg:mx-2.5">
      <span className="absolute opacity-30">{children}</span>
      <motion.span style={{ opacity }} className="text-white">{children}</motion.span>
    </span>
  )
}
```

```tsx
'use client'
import { useEffect, useState } from 'react'
import { Command } from 'cmdk'
import { motion, AnimatePresence } from 'framer-motion'

export interface CommandPaletteProps {
  onRun?: () => void
  onPause?: () => void
  onSave?: () => void
  onToggleConsole?: () => void
  onFitView?: () => void
}

export function CommandPalette({ onRun, onPause, onSave, onToggleConsole, onFitView }: CommandPaletteProps): JSX.Element {
  const [open, setOpen] = useState(false)

  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') { e.preventDefault(); setOpen(o => !o) }
      if ((e.metaKey || e.ctrlKey) && e.key === 's') { e.preventDefault(); onSave?.() }
      if ((e.metaKey || e.ctrlKey) && e.key === '`') { e.preventDefault(); onToggleConsole?.() }
    }
    window.addEventListener('keydown', handler)
    return () => window.removeEventListener('keydown', handler)
  }, [onSave, onToggleConsole])

  return (
    <AnimatePresence>
      {open && (
        <>
          <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} className="fixed inset-0 bg-black/60 backdrop-blur-sm z-50" onClick={() => setOpen(false)} />
          <motion.div initial={{ opacity: 0, scale: 0.95 }} animate={{ opacity: 1, scale: 1 }} exit={{ opacity: 0, scale: 0.95 }} className="fixed left-1/2 top-1/3 -translate-x-1/2 z-50 w-[560px]">
            <Command className="bg-[#0a0a0a] border border-white/10 rounded-2xl shadow-2xl overflow-hidden">
              <Command.Input placeholder="Search commands..." className="bg-transparent text-white placeholder:text-white/30 text-sm px-4 py-3 outline-none border-b border-white/[0.08] w-full" />
              <Command.List className="p-2 max-h-[400px] overflow-y-auto">

                  {onPause && <Command.Item onSelect={() => { onPause(); setOpen(false) }} className="px-3 py-2 text-sm text-white/70 hover:bg-white/[0.05] hover:text-white rounded-lg cursor-pointer">⏸ Pause Workflow</Command.Item>}
```

```typescript
import { useCallback } from 'react'
import type { Edge } from '@xyflow/react'

/** Returns true if adding the given edge would create a cycle in the graph. */
function hasCycle(edges: Edge[], newEdge: { source: string; target: string }): boolean {
  // Build adjacency list
  const adj: Record<string, string[]> = {}
  for (const e of [...edges, newEdge]) {
    if (!adj[e.source]) adj[e.source] = []
    adj[e.source]!.push(e.target)
  }

  // DFS from newEdge.source to see if we can reach newEdge.source again via newEdge.target
  const visited = new Set<string>()
  const stack = [newEdge.target]
  while (stack.length > 0) {
    const node = stack.pop()!
    if (node === newEdge.source) return true
    if (visited.has(node)) continue
    visited.add(node)
    for (const neighbor of adj[node] ?? []) stack.push(neighbor)
  }
  return false
}

export function useDagValidation() {
  const validateEdge = useCallback((edges: Edge[], newEdge: { source: string; target: string }): { valid: boolean; reason?: string } => {
    if (newEdge.source === newEdge.target) return { valid: false, reason: 'Self-loops are not allowed.' }
    if (hasCycle(edges, newEdge)) return { valid: false, reason: 'Cycles are not allowed in this workflow.' }
    return { valid: true }
  }, [])

  return { validateEdge }
}
```

```typescript
'use client'
import { useState, useCallback, useRef } from 'react'
import { useMutation, useQuery, gql } from '@apollo/client'
import type { Node, Edge } from '@xyflow/react'

const GET_WORKFLOW = gql`
  query GetWorkflow($id: ID!) {
    workflow(id: $id) { id name definition updatedAt }
  }
`

const UPDATE_WORKFLOW = gql`
  mutation UpdateWorkflow($id: ID!, $input: UpdateWorkflowInput!) {
    updateWorkflow(id: $id, input: $input) { id updatedAt }
  }
`

export type SaveStatus = 'idle' | 'saving' | 'saved' | 'error'

export function useWorkflow(workflowId: string) {
  const [saveStatus, setSaveStatus] = useState<SaveStatus>('idle')
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null)

  const { data, loading } = useQuery(GET_WORKFLOW, {
    variables: { id: workflowId },
    skip: !workflowId || workflowId === 'new',
  })

  const [updateWorkflow] = useMutation(UPDATE_WORKFLOW)

  const save = useCallback(async (name: string, nodes: Node[], edges: Edge[]) => {
    if (!workflowId || workflowId === 'new') return
    setSaveStatus('saving')
    try {
      await updateWorkflow({
        variables: {
          id: workflowId,
          input: { name, definition: { nodes, edges } },
        },
      })
      setSaveStatus('saved')
      setTimeout(() => setSaveStatus('idle'), 2000)
    } catch {
      setSaveStatus('error')
    }
  }, [workflowId, updateWorkflow])

  const debouncedSave = useCallback((name: string, nodes: Node[], edges: Edge[]) => {
    if (debounceRef.current) clearTimeout(debounceRef.current)
    debounceRef.current = setTimeout(() => save(name, nodes, edges), 1500)
  }, [save])

  const workflow = data?.workflow
  const initialNodes: Node[] = workflow?.definition?.nodes ?? []
  const initialEdges: Edge[] = workflow?.definition?.edges ?? []

  return { workflow, loading, initialNodes, initialEdges, saveStatus, debouncedSave }
}
```

```typescript
'use client'
import { useState, useCallback } from 'react'
import { useMutation, useSubscription, gql } from '@apollo/client'

const CREATE_RUN = gql`mutation CreateRun($workflowId: ID!) { createRun(workflowId: $workflowId) { id status } }`
const PAUSE_RUN = gql`mutation PauseRun($runId: ID!) { pauseRun(runId: $runId) { id status } }`
const RESUME_RUN = gql`mutation ResumeRun($runId: ID!, $decision: String!) { resumeRun(runId: $runId, decision: $decision) { id status } }`

const RUN_UPDATED = gql`subscription WorkflowRunUpdated($runId: ID!) { workflowRunUpdated(runId: $runId) { id status startedAt endedAt } }`
const NODE_EXEC_UPDATED = gql`subscription NodeExecutionUpdated($runId: ID!) { nodeExecutionUpdated(runId: $runId) { id nodeId status input output startedAt endedAt } }`
const NODE_LOG_STREAM = gql`subscription NodeLogStream($runId: ID!, $nodeId: ID!) { nodeLogStream(runId: $runId, nodeId: $nodeId) { token timestamp } }`

export function useWorkflowRun(workflowId: string) {
  const [runId, setRunId] = useState<string | null>(null)
  const [runStatus, setRunStatus] = useState<string | null>(null)
  const [nodeExecutions, setNodeExecutions] = useState<Record<string, { status: string; startedAt?: string; endedAt?: string }>>({})
  const [logs, setLogs] = useState<string[]>([])
  const [humanGatePending, setHumanGatePending] = useState(false)

  const [createRunMutation] = useMutation(CREATE_RUN)
  const [pauseRunMutation] = useMutation(PAUSE_RUN)
  const [resumeRunMutation] = useMutation(RESUME_RUN)

  const createRun = useCallback(async () => {
    const { data } = await createRunMutation({ variables: { workflowId } })
    if (data?.createRun?.id) {
      setRunId(data.createRun.id)
      setRunStatus(data.createRun.status)
      setLogs([])
      setNodeExecutions({})
    }
  }, [workflowId, createRunMutation])

  const pauseRun = useCallback(async () => {
    if (!runId) return
    await pauseRunMutation({ variables: { runId } })
  }, [runId, pauseRunMutation])

  const resumeRun = useCallback(async (decision: string) => {
    if (!runId) return
    await resumeRunMutation({ variables: { runId, decision } })
    setHumanGatePending(false)
  }, [runId, resumeRunMutation])

  // Subscriptions (only active when runId is set)
  useSubscription(RUN_UPDATED, {
    variables: { runId },
    skip: !runId,
    onData: ({ data }) => {
      const run = data.data?.workflowRunUpdated
      if (run) setRunStatus(run.status)
    },
  })

  useSubscription(NODE_EXEC_UPDATED, {
    variables: { runId },
    skip: !runId,
    onData: ({ data }) => {
      const ne = data.data?.nodeExecutionUpdated
      if (ne) {
        setNodeExecutions(prev => ({ ...prev, [ne.nodeId]: { status: ne.status, startedAt: ne.startedAt, endedAt: ne.endedAt } }))
        if (ne.status === 'PAUSED') setHumanGatePending(true)
      }
    },
  })

  return { runId, runStatus, nodeExecutions, logs, humanGatePending, createRun, pauseRun, resumeRun, setLogs }
}
```

```tsx
import { useRef, useEffect } from 'react'

const NODE_LOG_STREAM = gql`subscription NodeLogStream($runId: ID!, $nodeId: ID!) { nodeLogStream(runId: $runId, nodeId: $nodeId) { token timestamp } }`

const STATUS_COLOR: Record<string, string> = {
  RUNNING: 'text-cyan-400 border-cyan-500/30 bg-cyan-500/10',
  SUCCESS: 'text-green-400 border-green-500/30 bg-green-500/10',
  FAILED: 'text-red-400 border-red-500/30 bg-red-500/10',
  FALLBACK: 'text-yellow-400 border-yellow-500/30 bg-yellow-500/10',
  PENDING: 'text-white/30 border-white/10 bg-white/[0.03]',
}

interface ExecutionConsoleProps {
  runId: string
  selectedNodeId: string | null
  onClose: () => void
}

export function ExecutionConsole({ runId, selectedNodeId, onClose }: ExecutionConsoleProps): JSX.Element {
  const logsEndRef = useRef<HTMLDivElement>(null)
  const { nodeExecutions, logs, setLogs, runStatus } = useWorkflowRun('')

  useSubscription(NODE_LOG_STREAM, {
    variables: { runId, nodeId: selectedNodeId ?? '' },
    skip: !runId || !selectedNodeId,
    onData: ({ data }) => {
      const token = data.data?.nodeLogStream?.token
      if (token) setLogs(prev => [...prev, token])
    },
  })

  useEffect(() => {
    logsEndRef.current?.scrollIntoView({ behavior: 'smooth' })
  }, [logs])

  return (
    <motion.div
      initial={{ y: '100%' }}
      animate={{ y: 0 }}
      exit={{ y: '100%' }}
      transition={{ type: 'spring', stiffness: 300, damping: 30 }}
      className="absolute bottom-0 left-0 right-0 z-30 h-64 bg-black/90 backdrop-blur-md border-t border-white/[0.08] flex flex-col"

      {/* Top bar */}
      <div className="flex items-center justify-between px-4 py-2 border-b border-white/[0.08]">
        <div className="flex items-center gap-3">
          <span className="text-white/40 text-xs font-mono">▼ Execution Console</span>
          <span className="text-white/20 text-xs font-mono">Run #{runId.slice(0, 8)}</span>
          {runStatus && (
            <span className={cn('text-xs px-2 py-0.5 rounded border font-mono', STATUS_COLOR[runStatus] ?? STATUS_COLOR['PENDING'])}>
              {runStatus}
            </span>
          )}
        </div>
        <div className="flex items-center gap-2">
          <button onClick={() => setLogs([])} className="text-white/30 hover:text-white/60 text-xs transition-colors">Clear</button>
          <button onClick={onClose} className="text-white/30 hover:text-white/60 transition-colors"><X className="w-3.5 h-3.5" /></button>
        </div>
      </div>

      {/* Node pills row */}
      <div className="flex items-center gap-2 px-4 py-2 border-b border-white/[0.06] overflow-x-auto">
        {Object.entries(nodeExecutions).map(([nodeId, exec]) => (
          <span key={nodeId} className={cn('text-xs px-2 py-0.5 rounded border font-mono whitespace-nowrap cursor-pointer', STATUS_COLOR[exec.status] ?? STATUS_COLOR['PENDING'])}>
            {nodeId.slice(0, 8)} {exec.status === 'RUNNING' ? '↻' : exec.status === 'SUCCESS' ? '✓' : exec.status === 'FAILED' ? '✗' : '…'}
          </span>
        ))}
      </div>

      {/* Log stream */}
      <div className="flex-1 overflow-y-auto px-4 py-3">
        {logs.length === 0 && <p className="text-white/20 text-xs font-mono">{selectedNodeId ? 'Waiting for logs...' : 'Select a node to view logs.'}</p>}
        {logs.map((token, i) => (
          <span key={i} className="text-white/70 text-xs font-mono">{token}</span>
        ))}
        <div ref={logsEndRef} />
      </div>
    </motion.div>
  )
}
```
